First Rule: If you want to use lambda, you must use vector. Don't use array.

Normal Lambda:

auto lambda_name = [&](parameters) -> return_type { function_body };


Recursive Lambda:

#include <iostream>

int main() {
    // 1. Define the lambda with 'auto&& self' as the first argument
    auto factorial = [&](auto&& self, int n) -> int { // Return type must be mentioned explicitly. In case of void function return type would be 'void'.         
        if (n <= 1) return 1;                           
        // 2. Call 'self' recursively, passing 'self' again
        return n * self(self, n - 1);
    };    

    // 3. Invoke by passing the lambda to itself
    int result = factorial(factorial, 5);
    
    std::cout << "Factorial of 5 is: " << result << std::endl;
    return 0;
}
